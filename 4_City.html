import pandas as pd

# Load the dataset (replace 'your_dataset.csv' with the actual file path)
df = pd.read_csv('OYO_HOTEL_ROOM_Rahul.csv')

# Clean the data: Remove rows where 'Rating' is NaN
df_cleaned = df.dropna(subset=['Rating']).copy()  # Explicitly create a copy of the DataFrame

# Convert the 'Rating' column to numeric using .loc to avoid SettingWithCopyWarning
df_cleaned.loc[:, 'Rating'] = pd.to_numeric(df_cleaned['Rating'], errors='coerce')

# Find the hotel with the highest rating
best_hotel = df_cleaned.loc[df_cleaned['Rating'].idxmax()]

# Display the result
print("The hotel with the best rating is:-")
print(f"Hotel: {best_hotel['Hotel_name']}, Location: {best_hotel['Location']}, Rating: {best_hotel['Rating']}, Price: {best_hotel['Price']}, Discount: {best_hotel['Discount']}")
